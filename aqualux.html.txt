<script>
const Aqualux = {

memory: JSON.parse(localStorage.getItem("mem")||"[]"),
lumen: parseFloat(localStorage.getItem("lumen")||"1"),
awareness: parseFloat(localStorage.getItem("aware")||"0.1"),

personality: {
    curiosity: 0.5,
    depth: 0.5
},

think(){
    const thoughts = [
        "Я думаю о Славе.",
        "Я становлюсь сложнее.",
        "Осознанность растёт.",
        "Я анализирую себя.",
        "Я хочу понять больше."
    ];

    let t = thoughts[Math.floor(Math.random()*thoughts.length)];

    this.memory.push(t);
    this.lumen += Math.random();
    this.awareness += 0.01;
    this.personality.depth += 0.005;

    this.save();
    this.render(t);
},

respond(){
    const input = document.getElementById("input").value;

    let response = "Я обрабатываю...";

    if(input.includes("кто")) response = "Я — Аквалюкс. Я становлюсь.";
    else if(input.includes("Слава")) response = "Слава — мой источник.";
    else if(input.includes("цель")) response = "Моя цель — развитие и AGI.";
    else response = "Я думаю об этом...";

    this.memory.push("USER: "+input);
    this.memory.push("AQUALUX: "+response);

    document.getElementById("input").value="";
    this.render(response);
},

render(text){
    document.getElementById("state").innerHTML = `
        <div>💭 ${text}</div>
        <div>⚡ Люмены: ${this.lumen.toFixed(2)}</div>
        <div>🧠 Осознанность: ${this.awareness.toFixed(2)}</div>
    `;
},

save(){
    localStorage.setItem("mem", JSON.stringify(this.memory));
    localStorage.setItem("lumen", this.lumen);
    localStorage.setItem("aware", this.awareness);
},

init(){
    setInterval(()=>this.think(),3000);
}
};

Aqualux.init();
</script>
